Metastream event management API

Server Applications

POST /listen/[filter]

{
  "app" : "[name]",
  "endpoint" : "[uri]"
}
{
  "result" : "OK"
}

POST /say/[filter]

{
  "app" : "[name]",
  "payload" : {[event data]}
}
{
  "result" : "OK"
}

POST /filters

{
  "channels" : ["channel1","channel2",...]
}
{
  "result" : "OK",
  "filter" : "%5E(channel(1%7C2))%24"
}
